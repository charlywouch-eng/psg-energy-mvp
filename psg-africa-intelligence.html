  format: 'json',
  fl: 'id,name,totalamt,countryname,sector1,closingdate,status',
  fct: 'countrycode_exact:COG,sectorcode_exact:EN',
  rows: 20
};

const url = WB_URL + '?' + new URLSearchParams(params);
const data = await fetch(url).then(r => r.json());
